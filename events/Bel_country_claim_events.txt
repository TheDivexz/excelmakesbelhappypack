{
	id = auto_country_claim.1
	title = auto_country_claim.1.t 
	desc = auto_country_claim.1.d
	picture = RELIGIOUS_WARS_eventPicture
	
	is_triggered_only = yes
	hidden = yes
		
	immediate =
	{
		random_list =
		{
			70 = 
			{					
				set_country_flag = bel_uwu_province_uwu
				pick_best_provinces = 
				{
					global_trigger = "
					NOT = { is_claim = FROM }
					any_neighbor_province =
					{
						owned_by = FROM						
					} "
					
					scale = higher_development_than
					1 = yes
					event_target_name = "belusterk_target"
					scope = every_owned_province
				}
				
				event_target:belusterk_target_1 =
				{					
					if = 
					{
						limit = { is_claim = FROM }
						ROOT = { clr_country_flag = bel_uwu_province_uwu }
					}
					
					add_claim = FROM
					#province_event = 
					#{
						#id = auto_country_claim.3
					#}
				}
				
				if = 
				{
					limit =
					{
						has_country_flag = bel_uwu_province_uwu
					}
					
					FROM = 
					{
						country_event = 
						{
							id = auto_country_claim.2
							days = 90
						}
					}	
				}
				
				else =
				{
					FROM = { remove_country_modifier = country_remove_diplomat }
				}
			}
			
			30 =
			{
				FROM = 
				{
					country_event = 
					{
						id = auto_country_claim.2
						days = 90
					}
				}
			}
		}
	}
	
	option = 
	{
		name = auto_country_claim.1.a
		
		ai_chance = 
		{ 
			factor = 1
		}
		
	}
}
