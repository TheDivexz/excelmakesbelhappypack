country_decisions = {
	dev_click = {
		major = yes
		
		potential = {
			ai = yes
			REB = { check_variable = { which = dev_click_max	value = 600 } }
		}

		allow = {
			custom_trigger_tooltip = {
				tooltip = monarch_power_trigger.tt
				has_country_flag = HitPointCap
			}
		}
		
		provinces_to_highlight = {
			OR = {
				has_province_flag = CheapDevTargetFlag 
				has_province_flag = CheapDipTargetFlag 
			}
			owned_by = ROOT
		}
		
		effect = {
			clr_country_flag = HitPointCap
			custom_tooltip = increase_dev_event.2.a.ct
			every_owned_province = {
				limit = {
					OR = {
						has_province_flag = CheapDevTargetFlag 
						has_province_flag = CheapDipTargetFlag 
					}
				}
				custom_tooltip = increase_dev_event.2.b.ct
			}
			hidden_effect = {
				# export the current monarch points
				export_to_variable = { variable_name = adm_var	value = trigger_value:adm_power }
				export_to_variable = { variable_name = mil_var	value = trigger_value:mil_power }
				export_to_variable = { variable_name = dip_var	value = trigger_value:dip_power	}
				if = {
					limit = {
						check_variable = { which = adm_var	which = mil_var }
						check_variable = { which = adm_var	which = dip_var }
					}
					country_event = { id = increase_dev_event.3 }
				}
				else_if = {
					limit = {
						check_variable = { which = dip_var	which = mil_var }
					}
					country_event = { id = increase_dev_event.4 }
				}
				else = { 
					country_event = { id = increase_dev_event.5 }
				}
			}
		}

		ai_will_do = {
			factor = 1
		}
	}
}
