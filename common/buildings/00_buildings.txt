wharf = 
{
	manufactory = 
	{
		naval_supplies
		fish
		salt
	}
	
	onmap = yes
	on_built = 
	{
		add_development_from_building = 
		{
			building = manufactory
			type = production
			value = 2
		}
	}
	
	on_destroyed = 
	{
		remove_development_from_building = 
		{
			building = manufactory
			type = production
			value = 2
		}
	}

	ai_will_do = 
	{
		factor = 1

		modifier = 
		{
			factor = 0.4
			NOT = 
			{ 
				base_production = 6 
			}
		}
		
		modifier = 
		{
			factor = 1.5
			base_production = 10
		}
		
		modifier = 
		{
			factor = 1.5
			base_production = 15
		}
		
		modifier = 
		{
			factor = 5
			OR = 
			{
				has_province_modifier = skanemarket
				has_province_modifier = comet_ore_modifier
				has_province_modifier = paradise_modifier
			}
		}
		
		modifier = 
		{
			NOT = 
			{ 
				num_free_building_slots = 3 
			}
			
			factor = 1.5
		}
	}
}
