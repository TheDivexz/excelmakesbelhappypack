sripted_trigger_loc_dev_cost_less_than = {
	#checks if value is lower than the devcost
	variable_arithmetic_trigger = {
		export_to_variable = {
			variable_name = varibale	
			value = modifier:local_development_cost
		} #import mod
		NOT = {
			check_variable = {
				which = varibale
				value = $value$
			}
		}
	}
}
